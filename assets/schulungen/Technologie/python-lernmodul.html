import smtplib from email.mime.multipart import MIMEMultipart from email.mime.text import MIMEText from email.mime.base import MIMEBase from email import encoders def sende_bericht(empfaenger: str, betreff: str, text: str, anhang: str = None): msg = MIMEMultipart() msg["From"] = "bericht@tommy-moeller.de" msg["To"] = empfaenger msg["Subject"] = betreff msg.attach(MIMEText(text, "html")) if anhang: with open(anhang, "rb") as f: teil = MIMEBase("application", "octet-stream") teil.set_payload(f.read()) encoders.encode_base64(teil) teil.add_header("Content-Disposition", f'attachment; filename="{anhang}"') msg.attach(teil) with smtplib.SMTP_SSL("smtp.gmail.com", 465) as server: server.login("user@gmail.com", "app-passwort") server.send_message(msg) # Automatischer Monatsbericht sende_bericht( "chef@unternehmen.de", "Portfolio-Report Mai 2025", "<h2>Anbei der monatliche Report</h2>", anhang="portfolio_report.xlsx" )

Vollständiger Monatsreport-Generator

Python – Vollständiges Skript
import pandas as pd from datetime import datetime def erstelle_monatsreport(csv_pfad: str, monat: int, jahr: int) -> str: """Liest Zahlungsdaten und erstellt Excel-Monatsbericht.""" df = pd.read_csv(csv_pfad, parse_dates=["datum"]) # Monatsfilter mask = (df["datum"].dt.month == monat) & (df["datum"].dt.year == jahr) df_monat = df[mask].copy() # KPIs berechnen kpis = { "Anzahl Zahlungen": len(df_monat), "Summe Eingänge": df_monat["betrag"].sum(), "Durchschnitt": df_monat["betrag"].mean(), "Höchste Zahlung": df_monat["betrag"].max(), "Rückstände": len(df_monat[df_monat["status"] == "offen"]), } # Segment-Auswertung seg_pivot = df_monat.groupby("segment")["betrag"].agg( ["count", "sum", "mean"] ).rename(columns={"count":"Anzahl","sum":"Summe","mean":"Durchschnitt"}) # In Excel schreiben dateiname = f"report_{jahr}_{monat:02d}.xlsx" with pd.ExcelWriter(dateiname, engine="openpyxl") as writer: pd.DataFrame([kpis]).to_excel(writer, sheet_name="KPIs", index=False) seg_pivot.to_excel(writer, sheet_name="Segmente") df_monat.to_excel(writer, sheet_name="Rohdaten", index=False) print(f"✓ Report gespeichert: {dateiname}") return dateiname # Ausführen erstelle_monatsreport("zahlungen.csv", monat=5, jahr=2025)

Wann Python vs. Excel?

Python besser

Wähle Python wenn...

  • Mehr als 100.000 Zeilen Daten
  • Wiederkehrende Aufgabe (Automatisierung)
  • Mehrere Dateien zusammenführen
  • API-Anbindung nötig
  • Komplexe Transformationslogik
  • Versionskontrolle (Git) gewünscht
Excel besser

Wähle Excel wenn...

  • Einmalige Ad-hoc-Analyse
  • Kollegen ohne Python-Kenntnisse
  • Interaktive Pivot-Analyse
  • Charts für Präsentationen
  • Kleines Datenvolumen (<50k Zeilen)
🎯 Ideale Kombination: Python für Datenabruf, Bereinigung und Aggregation – dann Export als Excel für die manuelle Review und Präsentation. Das Beste aus beiden Welten.

Wissenscheck Python

7 Fragen zu Python-Grundlagen und pandas.

1. Was ist ein pandas DataFrame?
Ein DataFrame ist die zentrale Datenstruktur von pandas – tabellarische Daten mit Zeilen und Spalten, ähnlich wie Excel oder eine SQL-Tabelle, aber mit Python-Operationen.
2. Was macht df.groupby("segment").agg({"volumen": "sum"})?
groupby() kombiniert mit agg() ist das Python-Äquivalent von SQL's GROUP BY mit Aggregationsfunktionen. Hier: Volumen-Summe je Segment.
3. Wie liest du eine Excel-Datei "report.xlsx" in einen DataFrame?
pd.read_excel() liest Excel-Dateien. pd.read_csv() ist für CSV-Dateien. Zusätzlich kann man sheet_name angeben, um ein bestimmtes Tabellenblatt zu laden.
4. Was ist der Unterschied zwischen & und "and" in pandas-Filtern?
Bei pandas-Filtern mit mehreren Bedingungen muss & statt "and" verwendet werden, da pandas mit Arrays (Series) arbeitet. Außerdem müssen Bedingungen in Klammern stehen: (df["a"] > 0) & (df["b"] == "x").
5. Was ist ein f-String in Python?
f-Strings (Format-Strings) sind die modernste Art der String-Formatierung in Python. Mit f"Hallo {name}, Volumen: {v:,.2f} €" werden Variablen direkt in Strings eingebettet.
6. Wann ist Python Excel klar überlegen?
Python glänzt bei Automatisierung (einmal schreiben, immer laufen lassen), großen Datenmengen, API-Integration und komplexen Transformationen. Excel bleibt besser für interaktive Analysen und Kollaboration ohne Programmierkenntnisse.
7. Was gibt df.isnull().sum() aus?
df.isnull() gibt eine Boolean-Maske zurück (True = fehlend), .sum() zählt die True-Werte pro Spalte. Das ist der schnellste Weg zur Datenqualitätsprüfung.
